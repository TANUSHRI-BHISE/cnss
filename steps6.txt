üß© PART 3 ‚Äî Create UDP Client & Server
üñ•Ô∏è Server (Receiver)

Make a file called udp_server.py:

import socket

IP = "0.0.0.0"   # Listen on all interfaces
PORT = 5001

sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
sock.bind((IP, PORT))
print(f"üì° Server listening on {IP}:{PORT}...")

# Receive file info
filename, addr = sock.recvfrom(1024)
filename = filename.decode()
print(f"Receiving file: {filename} from {addr}")

with open("received_" + filename, "wb") as f:
    while True:
        data, addr = sock.recvfrom(4096)
        if data == b"EOF":
            break
        f.write(data)

print(f"‚úÖ File '{filename}' received successfully.")
sock.close()

üíª Client (Sender)

Make a file called udp_client.py:

import socket, os

SERVER_IP = "127.0.0.1"  # Change to Server PC IP if using 2 PCs
PORT = 5001

sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

files = ["sample.txt", "music.mp3", "video.mp4", "script.py"]

def send_file(filename):
    print(f"üöÄ Sending file: {filename}")
    sock.sendto(filename.encode(), (SERVER_IP, PORT))
    with open(filename, "rb") as f:
        data = f.read(4096)
        while data:
            sock.sendto(data, (SERVER_IP, PORT))
            data = f.read(4096)
    sock.sendto(b"EOF", (SERVER_IP, PORT))
    print(f"‚úÖ Sent '{filename}' successfully!\n")

for file in files:
    if os.path.exists(file):
        send_file(file)
    else:
        print(f"‚ö†Ô∏è {file} not found, skipping...")

sock.close()
